💡 Como funciona a randomização

Quando você seleciona múltiplos números e gera o link, um único link é gerado.
Cada lead que clicar será direcionado aleatoriamente para um dos números selecionados.
O sistema registra qual número recebeu cada lead — visível na aba Leads.
Limite: até 20 números cadastrados.

Instruções de integração

1. Script para o <head> da pretzel

Captura o click ID do Kwai automaticamente e atualiza os links:

<script> (function(){ function gp(n){var r=new RegExp('[?&]'+n+'=([^&#]*)').exec(location.href);return r?decodeURIComponent(r[1]):null;} var cid=gp('clickid')||gp('kwai_click_id')||gp('kpf')||gp('click_id'); if(cid){ var e=new Date();e.setDate(e.getDate()+7); document.cookie='kwai_cid='+cid+';expires='+e.toUTCString()+';path=/'; try{localStorage.setItem('kwai_cid',cid);}catch(x){} } document.addEventListener('DOMContentLoaded',function(){ function gc(n){var m=document.cookie.match(new RegExp('(^| )'+n+'=([^;]+)'));return m?m[2]:null;} var id=gc('kwai_cid')||(function(){try{return localStorage.getItem('kwai_cid')}catch(x){return null}})(); if(id){ document.querySelectorAll('a[href*="capi-track-production.up.railway.app/kwai/ir"]').forEach(function(a){ var url=new URL(a.href);url.searchParams.set('cid',id);a.href=url.toString(); }); } }); })(); </script>
